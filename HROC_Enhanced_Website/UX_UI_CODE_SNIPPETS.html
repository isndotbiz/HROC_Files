/**
 * Mobile hamburger menu with accessibility features
 */
function initMobileMenu() {
  const menuToggle = document.getElementById('menu-toggle');
  const menuItems = document.getElementById('menu-items');
  const submenuToggles = document.querySelectorAll('.submenu-toggle');

  if (!menuToggle || !menuItems) return;

  // Toggle menu
  menuToggle.addEventListener('click', () => {
    const isExpanded = menuToggle.getAttribute('aria-expanded') === 'true';

    menuToggle.setAttribute('aria-expanded', !isExpanded);
    menuItems.hidden = isExpanded;

    // Trap focus in menu when open
    if (!isExpanded) {
      const firstLink = menuItems.querySelector('a, button');
      if (firstLink) firstLink.focus();
    } else {
      menuToggle.focus();
    }
  });

  // Toggle submenus
  submenuToggles.forEach(toggle => {
    toggle.addEventListener('click', () => {
      const submenuId = toggle.getAttribute('aria-controls');
      const submenu = document.getElementById(submenuId);
      const isExpanded = toggle.getAttribute('aria-expanded') === 'true';

      toggle.setAttribute('aria-expanded', !isExpanded);
      submenu.hidden = isExpanded;
    });
  });

  // Close menu on Escape key
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && !menuItems.hidden) {
      menuToggle.setAttribute('aria-expanded', 'false');
      menuItems.hidden = true;
      menuToggle.focus();
    }
  });

  // Close menu when clicking outside
  document.addEventListener('click', (e) => {
    if (!menuItems.hidden &&
        !menuToggle.contains(e.target) &&
        !menuItems.contains(e.target)) {
      menuToggle.setAttribute('aria-expanded', 'false');
      menuItems.hidden = true;
    }
  });
}

// Initialize on DOM ready
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', initMobileMenu);
} else {
  initMobileMenu();
}

/**
 * Service request form validation and submission
 */
function initServiceRequestForm() {
  const form = document.querySelector('.service-request-form');
  const errorsContainer = document.getElementById('form-errors');

  if (!form) return;

  form.addEventListener('submit', async (e) => {
    e.preventDefault();

    // Clear previous errors
    errorsContainer.hidden = true;
    errorsContainer.innerHTML = '';

    // Validate form
    const formData = new FormData(form);
    const errors = [];

    // Check urgency selected
    if (!formData.get('urgency')) {
      errors.push('Please select when you need help');
    }

    // Check contact info
    if (!formData.get('contact')) {
      errors.push('Please provide a way to reach you');
    }

    // If emergency selected, suggest calling instead
    if (formData.get('urgency') === 'emergency') {
      if (confirm('For emergency help, calling is faster. Would you like to call now?')) {
        window.location.href = 'tel:+12065551234';
        return;
      }
    }

    // Show errors if any
    if (errors.length > 0) {
      errorsContainer.innerHTML = `
        <strong>Please fix the following:</strong>
        <ul>
          ${errors.map(err => `<li>${err}</li>`).join('')}
        </ul>
      `;
      errorsContainer.hidden = false;
      errorsContainer.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
      return;
    }

    // Submit form
    try {
      const response = await fetch(form.action, {
        method: 'POST',
        body: formData
      });

      if (response.ok) {
        // Show success message
        form.innerHTML = `
          <div class="success-message" role="alert">
            <h2>‚úÖ Request Received</h2>
            <p><strong>We'll contact you within 2 hours</strong> at the contact you provided.</p>
            <div class="next-steps">
              <h3>While you wait:</h3>
              <ul>
                <li>üìç <a href="/locations">Find our nearest RV location</a></li>
                <li>üè• <a href="/naloxone">Learn about naloxone</a></li>
                <li>üìû Crisis? Call now: <a href="tel:+12065551234">(206) 555-1234</a></li>
              </ul>
            </div>
          </div>
        `;
      } else {
        throw new Error('Submission failed');
      }
    } catch (error) {
      errorsContainer.innerHTML = `
        <strong>Submission Error</strong>
        <p>We couldn't submit your request. Please call us directly at <a href="tel:+12065551234">(206) 555-1234</a>.</p>
      `;
      errorsContainer.hidden = false;
    }
  });
}

// Initialize
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', initServiceRequestForm);
} else {
  initServiceRequestForm();
}

Live Region Announcements

/**
 * Announce message to screen readers
 * @param {string} message - Message to announce
 * @param {string} priority - 'polite' (default) or 'assertive'
 */
function announceToScreenReader(message, priority = 'polite') {
  const announcement = document.createElement('div');
  announcement.setAttribute('role', 'status');
  announcement.setAttribute('aria-live', priority);
  announcement.className = 'sr-only';
  announcement.textContent = message;

  document.body.appendChild(announcement);

  // Remove after announcement (don't clutter DOM)
  setTimeout(() => {
    document.body.removeChild(announcement);
  }, 1000);
}

// Usage examples
announceToScreenReader('Form submitted successfully');
announceToScreenReader('Error: Please fix the highlighted fields', 'assertive');

Focus Management for Modals

/**
 * Trap focus within a modal
 * @param {HTMLElement} modal - Modal element
 */
function trapFocus(modal) {
  const focusableElements = modal.querySelectorAll(
    'a[href], button, textarea, input, select, [tabindex]:not([tabindex="-1"])'
  );
  const firstElement = focusableElements[0];
  const lastElement = focusableElements[focusableElements.length - 1];

  modal.addEventListener('keydown', (e) => {
    // Trap Tab key
    if (e.key === 'Tab') {
      if (e.shiftKey && document.activeElement === firstElement) {
        e.preventDefault();
        lastElement.focus();
      } else if (!e.shiftKey && document.activeElement === lastElement) {
        e.preventDefault();
        firstElement.focus();
      }
    }

    // Close on Escape
    if (e.key === 'Escape') {
      closeModal(modal);
    }
  });

  // Focus first element when modal opens
  firstElement.focus();
}

// Usage
const modal = document.getElementById('my-modal');
modal.removeAttribute('hidden');
trapFocus(modal);

Color Contrast Checker

/**
 * Calculate WCAG contrast ratio between two colors
 * @param {string} color1 - Hex color (e.g., '#0066cc')
 * @param {string} color2 - Hex color
 * @returns {number} Contrast ratio (1-21)
 */
function getContrastRatio(color1, color2) {
  const getLuminance = (hex) => {
    const rgb = parseInt(hex.slice(1), 16);
    const r = (rgb >> 16) & 0xff;
    const g = (rgb >> 8) & 0xff;
    const b = (rgb >> 0) & 0xff;

    const srgb = [r, g, b].map(val => {
      val /= 255;
      return val <= 0.03928 ? val / 12.92 : Math.pow((val + 0.055) / 1.055, 2.4);
    });

    return 0.2126 * srgb[0] + 0.7152 * srgb[1] + 0.0722 * srgb[2];
  };

  const lum1 = getLuminance(color1);
  const lum2 = getLuminance(color2);
  const brightest = Math.max(lum1, lum2);
  const darkest = Math.min(lum1, lum2);

  return (brightest + 0.05) / (darkest + 0.05);
}

// Usage
const ratio = getContrastRatio('#0066cc', '#ffffff');
console.log(`Contrast ratio: ${ratio.toFixed(1)}:1`);
